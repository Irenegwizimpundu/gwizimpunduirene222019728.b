        • SQL QUERIES

        1. Actors

          2.  CREATE TABLE Actors (
          Actor_Id INT PRIMARY KEY AUTO_INCREMENT,
          Name VARCHAR(100) NOT NULL,
          Email VARCHAR(100),
          Phone VARCHAR(20),
          Gender ENUM('Male', 'Female')
          );
        2. Directors

          3. CREATE TABLE Directors (
          Director_Id INT PRIMARY KEY AUTO_INCREMENT,
          Name VARCHAR(100) NOT NULL,
          Telephone VARCHAR(20,
          Email VARCHAR(100)
          );
        3. Film_Information

          4. CREATE TABLE Film_Information (
          Film_Info_Id INT PRIMARY KEY AUTO_INCREMENT,
          Actor_Id INT,
          Director_Id INT,
          Duration TIME,
          Title VARCHAR(255) NOT NULL,
          Release_Date DATE,
          FOREIGN KEY (Actor_Id) REFERENCES Actors(Actor_Id),
          FOREIGN KEY (Director_Id) REFERENCES Directors(Director_Id)
          );
        4. Financial_Management

          5. CREATE TABLE Financial_Management (
          Financial_Mgt_Id INT PRIMARY KEY AUTO_INCREMENT,
          Film_Info_Id INT,
          Budget DECIMAL(15, 2),
          Expenses DECIMAL(15, 2),
          Revenue DECIMAL(15, 2),
          Profit DECIMAL(15, 2),
          FOREIGN KEY (Film_Info_Id) REFERENCES Film_Info(Film_Info_Id)
          );
        5. Production_Details

          6. CREATE TABLE Production_Details (
          Production_Details_Id INT PRIMARY KEY AUTO_INCREMENT,
          Budget DECIMAL(10, 2),
          Marketing_Location VARCHAR(100),
          Production_Start_Date DATE,
          Production_End_Date DATE
          );
